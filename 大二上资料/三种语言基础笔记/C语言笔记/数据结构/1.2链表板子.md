# 链表板子
# 1.约瑟夫问题
```c
void josephu( int n, int k, int m )//n个人，第k个人开始报数，报到m出列 
{
	//构建一个n个节点的链表，数据为1-n 
	Nodeptr  p, q, list=NULL,r;
	int i=1;
    for(i=1;i<=n;i++)
    {
        q=(Nodeptr)malloc(sizeof(Node));
        q->data=i;     	/* 取一个数据元素 */ 
        q->link=NULL; 
        if (list==NULL) 	/*链表为空*/
            list=p=q;
        else
            p->link=q;       /* 将新结点链接在链表尾部 */
        p=q;
    }
    p->link=list;
    
	r=p;
	p=list;//初始化r为尾节点，p为头节点 
    for(p=list,i=1;i<=k-1;i++)
    {
    	r=p;
		p=p->link;
	}
    //此时p指向开始节点k
	while(p->link!=p)
	{
		for(i=1;i<=m-1;i++)
		{
			r=p;
			p=p->link; 
		}
		r->link=p->link;
		printf("%d ",p->data);
		free(p);
		p=r->link; 
	} 
	printf("%d",p->data); 
}
```
