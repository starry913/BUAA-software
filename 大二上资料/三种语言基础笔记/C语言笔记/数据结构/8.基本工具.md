# 基本工具
## 文本读入
```c
    FILE *in, *out; // 为读写文件定义文件指针
    in = fopen("input.txt", "r");
    out = fopen("output.txt", "w");

    if(in==NULL||out==NULL) 
    {
        printf ("Failed to open the file !\n");
        return 1;
    }

    while ((c = fgetc(in)) != EOF) // 从文件input.txt中依次读一个字符
    { 
        fputc(c, out); // 向文件output.txt中依次写一个字符
    }
    fgets(s,n,in);    //从文件in中读入一行(最多n-1个字符)，放入s中，返回s或NULL
    fputs(s,out);    //把s写入文件out
    fscanf(in,"%d",&score);    //从文件in中输入一个整数
    fprintf(out,"%d",score);    //输出一个整数给文件out

    fclose(in);
    fclose(out);
```
//文本读入单词
```c
int get_word(FILE *fp, char *dst) 
{
    int i = 0;
    char c;
    while ((c = fgetc(fp)) != EOF) {
        if (isalpha(c) != 0) {
            dst[i++] = tolower(c);
        } else if (i) {
            dst[i] = '\0';
            return 0;  // 正常情况，随便返回一个值
        }
    }
    return EOF;
}
```
