# 栈、队列
## 1.栈
```c
#define MAXSIZE 1000
ElemType  STACK[MAXSIZE];
int  Top=-1;

void  initStack( )
{
    Top= –1;
}
int  isEmpty( )
{
    return Top== –1;
} 
int isFull(  )
{
    return Top==MAXSIZE–1;
}

void  push( ElemType s[ ],  ElmeType item )//进栈
{
    if( isFull() )
        printf("Full Stack!");
    else
    {
        Top++;
        s[Top]=item;  
    }
}

ElemType pop( ElemType s[ ])    //弹栈
{
    if(isEmpty())
        printf("Empty Stack!");
    else
    {
        return s[Top--];
    }
             
}

```



## 2.队列
队尾插入，队头删除
```c
#define MAXSIZE 1000
#define ElemType  int
queue[MAXSIZE];
int  Front=0, Rear=MAXSIZE-1,Count=0;

void  initQueue(  ) 
{
    Front = 0;
    Rear = MAXSIZE-1;
    Count = 0;
}
int  isEmpty( )
{
    return Count == 0;
}
int  isFull( )
{
    return Count == MAXSIZE;
}

void enQueue(ElemType queue[ ], ElemType item)
{
    if(isFull())               /* 队满，插入失败 */  
        printf("Full queue!");
    else
    {
        Rear = (Rear+1) % MAXSIZE;        //循环队列
        queue[Rear]=item;
        Count++;
                        /* 队未满，插入成功 */ 
    }
}

ElemType  deQueue(ElemType queue[ ])
{ 
    ElemType e;
    if(isEmpty())
        printf("Empty queue!");     /* 队空，删除失败 */
    else
    {
        e=queue[Front];
        Count--;                             /* 队非空，删除成功 */
        Front = (Front+1)%MAXSIZE;        //循环队列
        return e;
    }
}
```
