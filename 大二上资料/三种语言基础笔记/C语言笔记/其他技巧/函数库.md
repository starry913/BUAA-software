```c
(int)round(x)    //四舍五入
(int)floor(x)    //向上取整
(int)ceil(x)     //向下取整
(int)(x)         //取整数部分
ans = (x + y - 1) / y;    // x/y 的向上取整(x,y都为整数)
ans = (x + y / 2) / y;    // x/y 的四舍五⼊取整(x,y都为整数)

if(a>=0) a=(int)(a+0.5);
```

# 2.字符处理函数(#include <ctype.h>)
```c
//以下判断函数，不成功返回 0 ，成功则返回⼀个⾮零数（不⼀定是1）
isdigit(ch)    //判断 ch 是否是数字字符（'0'~'9'）；
isalpha(ch)    //判断 ch 是否是字母（包括⼤⼩写）；
islower(ch)    //判断 ch 是否是⼩写字母；
isupper(ch)    //判断 ch 是否是⼤写字母；

//以下转换函数，符合条件转换，不符合条件原样输出
tolower(ch)    //⼤写转⼩写；
toupper(ch)    //⼩写转⼤写；
```
