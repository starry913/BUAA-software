```c
//最长回文串
#include <stdio.h>
#include <string.h>

void longestPalindromeSubstring(const char* s, char* result) {
    int n = strlen(s);
    if (n == 0) {
        result[0] = '\0';
        return;
    }

    // 创建dp数组
    int dp[n][n];
    memset(dp, 0, sizeof(dp));

    int start = 0, maxLength = 1;

    // 所有长度为1的子串都是回文
    for (int i = 0; i < n; ++i) {
        dp[i][i] = 1;
    }

    // 检查长度为2的子串
    for (int i = 0; i < n - 1; ++i) {
        if (s[i] == s[i + 1]) {
            dp[i][i + 1] = 1;
            start = i;
            maxLength = 2;
        }
    }

    // 检查长度大于2的子串
    for (int len = 3; len <= n; ++len) { // 子串长度
        for (int i = 0; i < n - len + 1; ++i) { // 子串起始位置
            int j = i + len - 1; // 子串结束位置
            if (s[i] == s[j] && dp[i + 1][j - 1]) {
                dp[i][j] = 1;
                start = i;
                maxLength = len;
            }
        }
    }

    // 提取最长回文子串
    strncpy(result, s + start, maxLength);
    result[maxLength] = '\0'; // 确保字符串以null终止
}

int main() {
    const char input[] = "babad";
    char result[100];
    longestPalindromeSubstring(input, result);
    printf("The longest palindromic substring is: %s\n", result);
    return 0;
}
```

```c
void strRev(char *s)
{
    char *plow = s, *phigh = s;
    while (*phigh)
    phigh++;
    phigh--; // phigh 指向 s 的最后一个字符
    while (plow < phigh)
    swap(plow++, phigh--);
}
void swap(char *plow, char *phigh)
{
    char tmp = *plow;
    *plow = *phigh;
    *phigh = tmp;
}
```

## 4.字符串排序

```c
//字符串长度排序（qsort 版）
char array[100][1001];
int lines[100]; // line number
int cmp(const void *e1, const void *e2)
{
    int i = *((int *) e1), j = *((int *) e2);//get line number
    int iLen = strlen(array[i]);
    int jLen = strlen(array[j]);
    if (iLen > jLen || iLen == jLen && strcmp(array[i], array[j]) > 0) return 1;

    if (iLen < jLen || iLen == jLen && strcmp(array[i], array[j]) < 0) return -1;
    return 0;
}
int main()
{
    int i, k = 0;
    while(gets(array[k]) != NULL)
    {
        lines[k] = k;//record line number
        k++;
    }
    qsort(lines, k, sizeof(int), cmp);
    for (i = 0; i < k; ++i)
    printf("%s\n", array[lines[i]]);
    return 0;
}
```

```c
void insert(char *s,int pos,char *str)//在字符串s的pos处插入字符串str
{
    int str_len = strlen(str);
    char tmp[600] = {0};
    strcpy(tmp,s+pos);
    strcpy(s+pos,str);

}
void delete(char *s,int pos,int n)//字符串s的pos处删除n个字符
{
    int str_len = strlen(s);
    if(pos+n > str_len){
        *(s+pos) = '\0';
        return ;
    }       
    char tmp[600] = {0};
    strcpy(tmp,s+pos+n);
    strcpy(s+pos,tmp);
}

    strcpy(s+pos,tmp);
}
```
