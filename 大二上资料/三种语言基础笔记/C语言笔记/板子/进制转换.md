# 进制转换
## 1.十进制$\Longrightarrow$K进制
```c
#include <stdio.h>
char b[9999999999];
int main()
{
	int a,i=1,j,r,k=2;
	scanf("%d",&a);
	while(a!=0)
	{
		r=a%k;
		if(r>=0&&r<=9) b[i]=r+'0';
		else b[i]=r-10+'A';
		a=a/k;
		i=i+1;
	}
	for(j=i-1;j>=1;j--) printf("%c",b[j]); 
 return 0;
}
```

## 2.K进制$\Longrightarrow$十进制
```c
#include <stdio.h>
#include<string.h>
int main()
{//假设输入的字符数组为k进制，a为转化后的十进制结果
	int a=0,k=16;
	int i; 
	char s[105];
	scanf("%s",s);
	int len=strlen(s);
	for(i=0;i<len;i++)
	{
		if(s[i]>='0'&&s[i]<='9') a=a*k+s[i]-'0';
		else a=a*k+s[i]-'A'+10;	
	} 
	printf("%d",a);
 return 0;
}
```
