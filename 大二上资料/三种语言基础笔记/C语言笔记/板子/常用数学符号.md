### 1.1判断素数
```c
//法1，基础遍历 范围 long long
int prime(long long num)
{
	if (num == 1)
	return 0;
	else if (num == 2 || num == 3) //特判5以内的素数
	return 1;
	else if (num % 6 != 5 && num % 6 != 1) //如果不在6的两侧
	return 0;
	else //在6的两侧，从5开始判断能否被素数整除，步长为6
	{
		for (long long i = 5; i * i <= num; i += 6)
		{
			if (num % i == 0 || num % (i + 2) == 0)
				return 0;
		}	
	return 1;
	}
}
```
```c
//法2，质数筛 范围:1e14，适合多次询问
#include <stdio.h>
#define MAX 10000005
int prime_map[MAX];
long long prime[MAX];
int cnt;
void init_prime() {  //初始化函数
   for (int i = 2; i < MAX; i++) prime_map[i] = 1;
   for (int i = 2; i * i < MAX; i++) {
       if (prime_map[i]) {
           for (int j = i * i; j < MAX; j += i)
               prime_map[j] = 0;
      }
  }
   for (int i = 0; i < MAX; i++) {
       if (prime_map[i]) prime[cnt++] = i;
  }
}
int is_prime(long long x) { //判断函数
   if (x < MAX)
       return prime_map[x];
   for (int i = 0; prime[i] * prime[i] <= x; i++) {
       if (x % prime[i] == 0)
           return 0;
  }
   return 1;
}
```

### 1.2最大公因数(辗转相除法)
**若求最大公倍数lcm,可用公式 lcm(a,b)=$\frac{a×b}{gcd(a,b)}$**
```c
int gcd(int a, int b)
{
	int r;
	if (b == 0)
	return a < 0 ? -a : a;
	while ((r = a % b) != 0)
	{
		a = b;
		b = r;
	}
	return b < 0 ? -b : b;
}
```

## 2.排列组合
### 2.1 组合数 O(min(m, n-m))
```c
long long com(int n, int k) {
    if (k > n - k) // 取较小的k值以减少循环次数
        k = n - k;
    
    long long result = 1;
    for (int i = 0; i < k; ++i) {
        result *= (n - i);
        result /= (i + 1); // 注意这里每次都要做除法以防止溢出
    }
    
    return result;
}
```
### 2.2 全排列
```c
#include <stdio.h>  // 引入标准输入输出库
// 全局变量声明
int n;  // 需要生成全排列的数字个数
int stack[10];  // 用于存储当前排列的数组
// 打印当前排列
void print_stack() {
    for (int i = 0; i < n; i++) {  // 遍历排列数组
        printf("%d", stack[i]);  // 打印当前数字
        if (i < n - 1) printf(" ");  // 如果不是最后一个数字，打印空格分隔
    }
    printf("\n");  // 换行
}
// 在当前排列中查找指定的数字
int find_in_stack(int key, int len) {
    for (int i = 0; i < len; i++)  // 遍历当前排列的前 len 个元素
        if (key == stack[i])  // 如果找到相同的数字
            return 1;  // 返回 1 表示找到
    return 0;  // 没有找到，返回 0
}
// 深度优先搜索函数
void dfs(int len) {
    if (len == n) {  // 如果当前排列长度达到 n
        print_stack();  // 打印当前排列
        return;  // 递归结束
    }
    for (int i = 1; i <= n; i++)  // 遍历从 1 到 n 的所有数字
        if (!find_in_stack(i, len)) {  // 如果当前数字不在当前排列中
            stack[len] = i;  // 将当前数字加入排列
            dfs(len + 1);  // 递归生成下一个位置的数字
        }
}
// 主函数
int main() {
    scanf("%d", &n);  // 从用户输入读取 n 的值
    dfs(0);  // 从第 0 个位置开始生成排列
    return 0;  // 程序结束
}
```

```c
#include <stdio.h>

int getWeek(int year, int month, int day);//查询某天是星期几
int isLeap(int year);//判断是否是闰年
int printYearCalender(int year, int weekday);// 打印年历
int getDaysOfMonth(int year, int month);// 获得月天数
int printMonthCalendar(int month, int days, int weekday);

int main()
{
	int year, weekday;
	printf("\ninput year: ");
	scanf("%d", &year);
	weekday = getWeek(year, 1, 1); // 求元旦是星期几
	printYearCalender(year, weekday); // 打印年历
	return 0;
}

int getWeek(int year,int month,int day) //查询某天是星期几
{
	int century,weekday; 
	// Zeller formula
	if(month<3)
	{
		year--;
		month += 12;
	}
	century = year/100;
	year = year%100;
	weekday = (year + year/4 + century/4 - 2*century +(26*(month+1))/10 + day - 1) % 7;
	if (weekday < 0)
	{
	weekday += 7;
	}
	return weekday;
}

int printYearCalender(int year, int weekday)// 打印年历
{
	int days, month;
	printf("\t\t\tyear: %d\n", year);
	for(month = 1; month <= 12; month++)
	{
		days = getDaysOfMonth(year, month);
		weekday = printMonthCalendar(month, days, weekday); // 返回值：下一个月的第一天是星期几
	}
	return weekday;
}

int printMonthCalendar(int month, int days, int weekday)
{
	int monthday, i;
	printf("\nmonth: %d\n", month);
	printf("------------------------------------------------------\n");
	printf("Sun\tMon\tTue\tWed\tThu\tFri\tSat\n");
	printf("------------------------------------------------------\n");
	for (i = 0; i < days + weekday; i++)
	i < weekday ? putchar('\t') : printf((i + 1) % 7 ? "%d\t" : "%d\n", i - weekday + 1);
	if (i % 7)
	putchar('\n');
	return i % 7; // 返回值：下一个月的第一天是星期几
}

int getDaysOfMonth(int year, int month) //返回值：某年某月有多少天
{
	int days = 0;
	switch(month)
	{
	case 1: case 3: case 5: case 7: case 8: case 10: case 12:
	days = 31;
	break;
	case 4: case 6: case 9: case 11:
	days = 30;
	break;
	case 2:
	days = isLeap(year) ? 29 : 28;
	break;
	}
	return days;
}

int isLeap(int year)
{
	return (year%4 == 0) && (year%100 != 0) || (year%400 == 0); // 逢四则闰，百年不闰，四百再闰
}
```

### 4.1 二元一次方程
```c
#include <stdio.h>
#include <math.h>
#define eps 1e-6
int main()
{
	double a,b,c,delta,r1,r2;
	scanf("%lf%lf%lf",&a,&b,&c);
	if(a==0)
	{
		if(b==0)
		{
			if(c==0) printf("infinite solutions");
			else     printf("No real root");
		
		}
		else if(c!=0)         printf("%.2lf",-c/b);
		else                  printf("0.00");
	}
	else
	{
		delta=b*b-4*a*c;
		if(fabs(delta)<=eps)
		{
			if(b!=0) printf("%.2lf",-b/(2*a));
			else     printf("0.00");
		}
		else if(delta>0.0)
		{
			r1=(-b-sqrt(b*b-4*a*c))/(2*a);
			r2=(-b+sqrt(b*b-4*a*c))/(2*a);
			if(a>0)	//r1<r2
			{
				if(r1==0) 		printf("0.00 %.2lf",r2);
				else if(r2==0)  printf("%.2lf 0.00",r1);
				else			printf("%.2lf %.2lf",r1,r2);
			}
			if(a<0) //r2<r1
			{
				if(r1==0) 		printf("%.2lf 0.00",r2);
				else if(r2==0)  printf("0.00 %.2lf",r1);
				else			printf("%.2lf %.2lf",r2,r1);
			}
		}
		else	printf("No real root");
	} 
 return 0;
}
```
