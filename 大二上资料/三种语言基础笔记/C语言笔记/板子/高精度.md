# 高精度算法
## 1.快速幂  $a^b\mod{c}$
```c
long long fastPowerMod(long long base, long long exponent, long long modulus) 
{
    long long result = 1;
    base = base % modulus;  // 确保base在模范围内
    while (exponent > 0) 
    {
        if (exponent & 1)  result=(result*base)%modulus;  // 若指数为1，把底数取出
        base=(base*base)%modulus;    // 底数乘方
        exponent>>=1;    // 右移1格
    }
    return result;
}
```

## 4.高精度四则运算
### 4.1高精度加法
```c
#include<stdio.h>
#include<string.h>
char s[10100],ss[10100];
int a[10100],b[10100],c[10100];
int len;
int main()
{
	scanf("%s%s",s,ss);
	int l1 = strlen(s);
	int l2 = strlen(ss);
	
	if (l1 > l2)  len = l1;//最大长度为len 
	else  len = l2;

	for (int i = l1 - 1 ; i >= 0 ; i--)//将字符串变为从低位到高位的int数组 
		a[l1 - i - 1] = s[i] - '0';
	for (int i = l2 - 1 ; i >= 0 ; i--)
		b[l2 - i - 1] = ss[i] - '0';
		
	for (int i = 0 ; i < len ; i++)//从低位数组相加 
	{
		if(c[i]+a[i]+b[i]<=9) c[i]=c[i]+a[i]+b[i];
		else 
		{
			c[i]=c[i]+a[i]+b[i]-10;
			c[i+1]++;	
		}
	}
	
	if (c[len] != 0) len++;
	while (c[len - 1] == 0 && len>1) len--;
	for (int i = len - 1 ;i >= 0 ;i--)//从高位输出 
		printf("%d",c[i]);
	return 0;
}
```

### 4.3高精度乘法
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>


char* multiply(char* num1, char* num2) 
{
    int len1 = strlen(num1);
    int len2 = strlen(num2);
    int result_len = len1 + len2;  
    
    // 用数组存储结果，初始化为0
    int* result = (int*)calloc(result_len, sizeof(int));
    

    for (int i = len1 - 1; i >= 0; i--) {      
        for (int j = len2 - 1; j >= 0; j--) {  

            int mul = (num1[i] - '0') * (num2[j] - '0');
            

            int pos1 = i + j;      
            int pos2 = i + j + 1;  
            

            int sum = mul + result[pos2];
            

            result[pos2] = sum % 10;    
            result[pos1] += sum / 10;   
        }
    }
    
    // 找到结果中第一个非零位，去除前导零
    int start = 0;
    while (start < result_len && result[start] == 0) {
        start++;
    }
    
    // 如果所有位都是0，说明结果是0
    if (start == result_len) {
        char* zero = (char*)malloc(2);
        zero[0] = '0';
        zero[1] = '\0';
        free(result);
        return zero;
    }
    
    // 将结果数组转换为字符串
    char* final_result = (char*)malloc(result_len - start + 1);
    int idx = 0;
    for (int i = start; i < result_len; i++) {
        final_result[idx++] = result[i] + '0';
    }
    final_result[idx] = '\0';
    
    free(result);
    return final_result;
}

int main() 
{
    int T;
    scanf("%d", &T);
    
    while (T--) {
        char A[2025], B[2025];
        scanf("%s", A);
        scanf("%s", B);
        
        char* result = multiply(A, B);
        printf("%s\n", result);
        free(result);
    }
    
    return 0;
}
```

### 4.4高精度除法
```c
#include <stdio.h>
#include <string.h>

// 比较两个大数的大小，返回1表示a>b，0表示a=b，-1表示a<b
int compare(char* a, char* b) {
    int len_a = strlen(a);
    int len_b = strlen(b);
    
    if (len_a > len_b) return 1;
    if (len_a < len_b) return -1;
    
    return strcmp(a, b);
}

// 高精度减法（被除数减去除数）
void subtract(char* a, char* b, char* result) {
    int len_a = strlen(a);
    int len_b = strlen(b);
    int borrow = 0;
    int i, j;
    
    // 从低位开始相减
    for (i = len_a - 1, j = len_b - 1; i >= 0; i--, j--) {
        int digit_a = a[i] - '0';
        int digit_b = (j >= 0) ? (b[j] - '0') : 0;
        
        digit_a -= borrow;
        if (digit_a < digit_b) {
            digit_a += 10;
            borrow = 1;
        } else {
            borrow = 0;
        }
        
        result[i] = (digit_a - digit_b) + '0';
    }
    
    // 去除前导零
    int start = 0;
    while (start < len_a && result[start] == '0') {
        start++;
    }
    
    if (start == len_a) {
        result[0] = '0';
        result[1] = '\0';
    } else {
        // 移动结果到开头
        for (i = 0; start < len_a; i++, start++) {
            result[i] = result[start];
        }
        result[i] = '\0';
    }
}

// 高精度除法
void divide(char* dividend, char* divisor, char* quotient, char* remainder) {
    int len_div = strlen(divisor);
    int len_dividend = strlen(dividend);
    
    // 特殊情况处理
    if (compare(dividend, divisor) < 0) {
        strcpy(quotient, "0");
        strcpy(remainder, dividend);
        return;
    }
    
    if (strcmp(divisor, "0") == 0) {
        strcpy(quotient, "ERROR");
        strcpy(remainder, "ERROR");
        return;
    }
    
    char temp[1005];
    char current[1005];
    strcpy(current, dividend);
    
    int q_pos = 0;
    int pos = 0;
    
    while (pos < len_dividend) {
        // 取当前位
        if (strlen(current) == 0) {
            current[0] = dividend[pos];
            current[1] = '\0';
        } else {
            int len = strlen(current);
            current[len] = dividend[pos];
            current[len + 1] = '\0';
        }
        
        // 去除前导零
        int start = 0;
        while (start < strlen(current) && current[start] == '0') {
            start++;
        }
        if (start == strlen(current)) {
            strcpy(current, "0");
        } else {
            for (int i = 0; start < strlen(current); i++, start++) {
                current[i] = current[start];
            }
            current[strlen(current) - (start - strlen(current))] = '\0';
        }
        
        // 试商
        int q = 0;
        char temp_divisor[1005];
        strcpy(temp_divisor, divisor);
        
        while (compare(current, temp_divisor) >= 0) {
            subtract(current, divisor, temp);
            strcpy(current, temp);
            q++;
        }
        
        quotient[q_pos++] = q + '0';
        pos++;
    }
    
    quotient[q_pos] = '\0';
    strcpy(remainder, current);
    
    // 去除商的前导零
    int start = 0;
    while (start < strlen(quotient) && quotient[start] == '0') {
        start++;
    }
    if (start == strlen(quotient)) {
        strcpy(quotient, "0");
    } else {
        for (int i = 0; start < strlen(quotient); i++, start++) {
            quotient[i] = quotient[start];
        }
        quotient[strlen(quotient) - (start - strlen(quotient))] = '\0';
    }
}

int main() {
    char dividend[1005], divisor[1005];
    char quotient[1005], remainder[1005];
    
    printf("请输入被除数: ");
    scanf("%s", dividend);
    printf("请输入除数: ");
    scanf("%s", divisor);
    
    divide(dividend, divisor, quotient, remainder);
    
    printf("商: %s\n", quotient);
    printf("余数: %s\n", remainder);
    
    return 0;
}
```

```c
// 高精度除法 - 返回商和余数
void high_precision_divide(char* dividend, char* divisor, char* quotient, char* remainder) {
    int len_dividend = strlen(dividend);
    int len_divisor = strlen(divisor);
    
    // 特殊情况：被除数小于除数
    if (len_dividend < len_divisor || 
        (len_dividend == len_divisor && strcmp(dividend, divisor) < 0)) {
        strcpy(quotient, "0");
        strcpy(remainder, dividend);
        return;
    }
    
    // 特殊情况：除数为0
    if (strcmp(divisor, "0") == 0) {
        strcpy(quotient, "ERROR");

        return;
    }
    
    char temp[1005];
    char current[1005];
    strcpy(current, "");
    
    int q_pos = 0;
    
    for (int i = 0; i < len_dividend; i++) {
        // 将当前位加入
        int len = strlen(current);
        current[len] = dividend[i];
        current[len + 1] = '\0';
        
        // 去除前导零
        int start = 0;
        while (start < strlen(current) && current[start] == '0') {
            start++;
        }
        if (start == strlen(current)) {
            strcpy(current, "0");
        } else {
            char temp2[1005];
            strcpy(temp2, current + start);
            strcpy(current, temp2);
        }
        
        // 试商
        int q = 0;
        char temp_div[1005];
        strcpy(temp_div, current);
        
        while (strlen(temp_div) > len_divisor || 
               (strlen(temp_div) == len_divisor && strcmp(temp_div, divisor) >= 0)) {
            // 模拟减法
            int borrow = 0;
            for (int j = len_divisor - 1, k = strlen(temp_div) - 1; j >= 0; j--, k--) {

                digit_temp -= borrow;

                    digit_temp += 10;
                    borrow = 1;
                } else {
                    borrow = 0;

                temp_div[k] = (digit_temp - digit_div) + '0';
            }
            
            // 去除前导零
            start = 0;
            while (start < strlen(temp_div) && temp_div[start] == '0') {
                start++;
            }
            if (start == strlen(temp_div)) {
                strcpy(temp_div, "0");

            strcpy(current, temp_div);

    quotient[q_pos] = '\0';
    strcpy(remainder, current);
    
    // 去除商的前导零

    while (start < strlen(quotient) && quotient[start] == '0') {
        start++;
    }
    if (start == strlen(quotient)) {
        strcpy(quotient, "0");

        strcpy(temp4, quotient + start);

}
```
