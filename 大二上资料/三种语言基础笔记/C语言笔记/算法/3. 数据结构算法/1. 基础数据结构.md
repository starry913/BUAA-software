# 基础数据结构
## 1.链表
l[i]表示 i节点左边节点的地址
r[i]表示 i节点右边节点的地址
data[i]表示 i节点的数据内容
```c
//双向链表
int l[N],r[N],data[N];
int head,tail;
void del(int x)//删去x
{
    if(r[x]==-1)
    {
        r[l[x]]=-1;
        tail=l[x];
    } 
    else
    {
        r[l[x]]=r[x];
        l[r[x]]=l[x];
    }
}
void insert_front(int y,int x)//x插在y前面
{
    r[x]=y;
    l[x]=l[y];
    l[r[x]]=x;
    r[l[x]]=x;
}
void insert_back(int y,int x)//x插在y后面
{
    if(r[y]==-1)
    {
        l[x]=y;
        r[x]=-1;
        r[l[x]]=x;
        tail=x;
    }
    else
    {
        l[x]=y;
        r[x]=r[y];
        l[r[x]]=x;
        r[l[x]]=x;
    }
}
```
## 2.堆(优先队列)
用途：**动态数组(可增加元素，可删去最值元素)求最值**<br>
父亲节点：a[x>>1]=a[x/2]   
左儿子：a[x<<1]=a[x×2]    
右儿子：a[x<<1|1]=a[x*2+1] 
```c
int a[300000000]; 
int size=0;
void swap(int *a,int *b)
{
	int c;
	c=*b;
	*b=*a;
	*a=c; 
}
int max(int a,int b,int c)
{
	if(a>=b&&a>=c) return a;
	if(b>=a&&b>=c) return b;
	return c;
} 
void pushup(int x)//将某个点向上更新 
{
	while(x!=1)
	{
		if(a[x]>a[x>>1]) swap(&a[x],&a[x>>1]),x=(x>>1);
		else break;
	}
}
void pushdown(int x)//将某个点向下更新 
{
	//如果没有儿子
	if((x<<1)>size) return;
	//如果只有左儿子 
	if((x<<1|1)>size)
	{
		if(a[x<<1]>a[x])
		{
			swap(&a[x],&a[x<<1]);
			pushdown(x<<1);
		}
	}
	//如果左右儿子都有
	else
	{
		if(a[x]==max(a[x],a[x<<1],a[x<<1|1])) return;
		if(a[x<<1]>a[x<<1|1])
		{
			swap(&a[x],&a[x<<1]);
			pushdown(x<<1);
		}
		else
		{
			swap(&a[x],&a[x<<1|1]);
			pushdown(x<<1|1);
		}
	} 
}
void push(int x)//插入一个点到堆里 
{
	a[++size]=x;
	pushup(size);
} 
void pop()//将根节点删除 
{
	a[1]=a[size--];
	pushdown(1);
}

int top()//取堆最大值
{
    return a[1];
}

void build_heap()//快速建堆(O(n))
{
    /* 如果 size==0 直接返回 */
    if (size <= 1) return;

    /* 完全二叉树中最后一个非叶节点是 size/2 */
    for (int i = size >> 1; i >= 1; --i)
        pushdown(i);   /* 你已有的函数 */
}
```

## 3.ST表
用途：**静态求区间最值**<br>
dp[i][j]表示从i开始，长度为$2^j$的区间元素最大值，即 [i, i+(2<<j)-1] 的最大值<br>
dp[i][j]=max(st[i][j-1], st[i+(1<<(j-1))][j-1]) 即max(左半边最大值，右半边最大值)<br>
先遍历j，再遍历i<br>
使用方法:
1. 求出dp数组
2. 将查询区间[l,r]分成两个长度为$2^k$的区间
   k=log2(r-l+1)
3. 答案为 max(st[l][k], st[r-(1<<k)+1][k])
```c
int a[100000],st[100000][21];
int max(int a,int b)
{
	if(a>b) return a;
	return b;
}
int getMax(int l,int r)
{
	int k=log(r-l+1)/log(2);
	return max(st[l][k],st[r-(1<<k)+1][k]);
}
int main()
{
	int n,m,i,j;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++) scanf("%d",&a[i]);
	
	//初始化
	for(i=1;i<=n;i++) st[i][0]=a[i];
	
	//先遍历长度j,再遍历开始位置i
	for(j=1;j<=20;j++)
		for(i=1;i<=n;i++)
		{
			//区间合法[i,i+(1<<j)-1] 
			if(i+(1<<j)-1<=n)
				st[i][j]=max(st[i][j-1],st[i+(1<<(j-1))][j-1]);
		}
	
	while(m--)
	{
		int l,r;
		scanf("%d%d",&l,&r);
		printf("%d\n",getMax(l,r));
		
	} 
    return 0;
}

```

## 4.并查集
用途:**存储图中结点的连通关系，可以判断两个点是否连通**
![alt text](360截图20250708235845894.jpg)
1.路径压缩法(降低时间复杂度，但会破坏图的父子关系)
```c
int pre[2000000];//pre[i]表示i指向的节点(一般称为父亲节点)
int root(int x)//路径压缩
{
	pre[x]=(pre[x]==x? x:root(pre[x]));
	return pre[x];
} 
int main()
{
	int n,m,i;
	scanf("%d%d",&n,&m);
	//初始化 
	for(i=1;i<=n;i++) pre[i]=i;
	while(m--)
	{
		int op,x,y;
		scanf("%d%d%d",&op,&x,&y);
		if(op==1)	//将x与y连通 
		{
			pre[root(x)]=root(y); //x的根节点指向y的根节点 
		}
		if(op==2) //查询x,y是否连通 
		{
			if(root(x)==root(y)) printf("YES\n");
			else printf("NO\n");
		}
	}
    return 0;
}
```
2.启发式合并(小集合指向大集合)
```c
int pre[2000000];//pre[i]表示i指向的节点(一般称为父亲节点)
int size[2000000];
int root(int x)
{
	return (pre[x]==x? x:root(pre[x]));
} 
void merge(int x,int y)//启发式合并两个点 
{
	int rx=root(x),ry=root(y);
	if(rx==ry)	return;
	if(size[rx]>=size[ry])
	{
		pre[ry]=rx;
		size[rx]+=size[ry]; 
	}
	else if(size[rx]<size[ry])
	{
		pre[rx]=ry;
		size[ry]+=size[rx]; 
	} 
}

int main()
{
	int n,m,i;
	scanf("%d%d",&n,&m);
	//初始化 
	for(i=1;i<=n;i++) pre[i]=i,size[i]=1;
	while(m--)
	{
		int op,x,y;
		scanf("%d%d%d",&op,&x,&y);
		if(op==1)	//将x与y连通 
		{
			merge(x,y);
		}
		if(op==2) //查询x,y是否连通 
		{
			if(root(x)==root(y)) printf("YES\n");
			else printf("NO\n");
		}
	}
    return 0;
}
```
## 5. 01Trie
时间复杂度：O(m*n)→O(m*logn)<br>
01Trie 是一种二叉树，每个叶子节点对应一个二进制数，每个树枝表示1个二进制位0/1;<br>
01Trie 支持三种操作<br>
1. 插入元素x
2. 查询01Trie中与x异或后的最大值，可解决最大异或对问题
3. 查询比x更大的元素个数，可解决逆序对问题
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>
#include <ctype.h>
int e[N];		//e[i]表示有多少元素经过i点 
int son[N][2];	//son[i][0]左儿子
int tot=2; 
void insert(int x)
{
	int o=1;
	e[o]++;
	for(int i=30;i>=0;i--)
	{
		int y=x>>i&1;	//y是x第i位的值
		if(!son[o][y]) son[o][y]=tot++;
		o=son[o][y];
		e[o]++;
	} 
}
//查询01Trie中与x异或后的最大值
int query(int x)
{
	int res=0,o=1;
	for(int i=30;i>=0;i--)
	{
		int y=x>>i&1;
		if(son[o][!y]) o=son[o][!y],res|=(1ll<<i);
		else o=son[o][y];
	}
	return res;
}
//查询比x更大的元素个数
int query(int x)
{
	int res=0,o=1;
	for(i=30;i>=0;i--)
	{
		int y=x>>i&1;
		if(y==0) res+=e[son[o][1]];
		if(!son[o][y]) break;
		o=son[o][y];
	}
	return res;
}
int main()
{
    int n,m,a,i,x;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a);
		insert(a);
	}
	for(i=1;i<=m;i++)
	{
		scanf("%d",&x);
		printf("%d\n",query(x));
	} 
    return 0;
}

```
