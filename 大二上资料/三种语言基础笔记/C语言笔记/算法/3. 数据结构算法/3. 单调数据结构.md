# 单调数据结构
## 1.单调栈
结构：内部元素具有单调性的栈(递增or递减)<br>
用途：求某个点左侧or右侧第一个比它大的点的位置<br>
实现方法：在入栈时逐个删除更差的点，保持单调性(栈顶为最优元素的位置)
```cpp
#include <bits/stdc++.h>
using namespace std;
const int N=2e6+9;
int h[N],dpl[N],dpr[N];//h[i]表示高度，dpl[i]表示i点左边第一个比它高的点的位置
int stk[N],top;
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&h[i]);
	//先处理dpl 
	for(int i=1;i<=n;i++)
	{
		while(top && h[stk[top]]<=h[i]) top--; //删除更差的点
		if(top) dpl[i]=stk[top];
		else    dpl[i]=-1;
		stk[++top]=i;               //存入当前点(当前点位置最优)
	}
	top=0;
	//再处理dpr 
	for(int i=n;i>=1;i--)
	{
		while(top && h[stk[top]]<=h[i]) top--;
		if(top) dpr[i]=stk[top];
		else    dpr[i]=-1;
		stk[++top]=i;
	}
	for(int i=1;i<=n;i++)printf("%d ",dpl[i]);
	printf("\n");
	for(int i=1;i<=n;i++)printf("%d ",dpr[i]);

    return 0;
}
```
## 2.单调队列
结构：内部元素具有单调性的双端队列(递增or递减)<br>
用途：求固定长度的区间最值<br>
实现方法：在入队时c从队尾逐个删除更差的点，保持单调性(队头为最优元素的位置)
```cpp
#include <bits/stdc++.h>
using namespace std;
const int N=2e6+9;
int a[N],q[N],mi[N],mx[N];//mi[i]表示以i点结尾，长度为k区间的最小值 

int main()
{
	int n,k;
	scanf("%d%d",&n,&k)
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	int hh=1,tt=0;
	//求固定长度区间k的最小值 
	for(int i=1;i<=n;i++)
	{
		//检查最优点(队头)是否合法
		while(hh<=tt&&q[hh]<i-k+1) hh++;
		//检查队尾是否能删除
		while(hh<=tt&&a[q[tt]]>a[i]) tt--;
		q[++tt]=i;
		mi[i]= a[q[hh]];
	}
	hh=1,tt=0;
	//求固定长度区间k的最大值 
	for(int i=1;i<=n;i++)
	{
		//检查最优点(队头)是否合法
		while(hh<=tt&&q[hh]<i-k+1) hh++;
		//检查队尾是否能删除
		while(hh<=tt&&a[q[tt]]<a[i]) tt--;
		q[++tt]=i;
		mx[i]= a[q[hh]];
	}

    return 0;
}

```
