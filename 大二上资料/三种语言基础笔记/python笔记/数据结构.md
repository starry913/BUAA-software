# 数据结构
## 1. 列表
**def:有序、可变、允许重复元素的集合**
### 1.1 创建列表
```py
# 创建空列表
empty_list = []
print(empty_list)

# 创建包含元素的列表
numbers = [1, 2, 3, 4, 5]
fruits = ["苹果", "香蕉", "橙子"]
mixed = [1, "Hello", 3.14, True]
print(numbers)
print(fruits)
print(mixed)

# 使用list()函数创建列表
string_list = list("Hello")
range_list = list(range(5))
print(string_list) # ['H', 'e', 'l', 'l', 'o']
print(range_list)  # [0, 1, 2, 3, 4]
```
### 1.2 访问列表元素
```py
fruits = ["苹果", "香蕉", "橙子", "葡萄", "西瓜"]

# 使用索引访问元素（索引从0开始）
print(fruits[0])  # 第一个元素
print(fruits[2])  # 第三个元素

# 使用负数索引（从末尾开始计数）
print(fruits[-1])  # 最后一个元素
print(fruits[-2])  # 倒数第二个元素

# 切片（获取子列表）
print(fruits[1:3])    # 索引1到2的元素  ['香蕉', '橙子']
print(fruits[:3])     # 从开始到索引2的元素
print(fruits[2:])     # 从索引2到末尾的元素
print(fruits[1:5:2])  # 索引1到4，步长为2的元素
print(fruits[::-1])   # 反转列表
```

### 1.3 修改列表
```py
fruits = ["苹果", "香蕉", "橙子"]
print("原始列表:", fruits)

# 修改单个元素
fruits[0] = "梨"
print("修改后:", fruits)

# 修改多个元素
fruits[1:3] = ["草莓", "蓝莓"]
print("修改后:", fruits)
```

### 1.4列表操作
```py
list1 = [1, 2, 3]
list2 = [4, 5, 6]

# 连接列表
combined = list1 + list2
print("连接后的列表:", combined)

# 重复列表
repeated = list1 * 3
print("重复后的列表:", repeated) #[1,2,3,1,2,3,1,2,3]

# 检查元素是否存在
print("2是否在list1中:", 2 in list1) #True
print("5是否在list1中:", 5 in list1) #False

# 列表长度
print("list1的长度:", len(list1))

# 最大值和最小值
numbers = [3, 1, 4, 1, 5, 9, 2, 6]
print("最大值:", max(numbers))
print("最小值:", min(numbers))
```

### 1.5列表方法
```py
fruits = ["苹果", "香蕉", "橙子"]
print("原始列表:", fruits)

# append() - 在列表末尾添加元素
fruits.append("葡萄")
print("append后:", fruits)

# insert() - 在指定位置插入元素
fruits.insert(1, "草莓")
print("insert后:", fruits)

# remove() - 删除指定元素
fruits.remove("香蕉")
print("remove后:", fruits)

# pop() - 删除并返回指定位置的元素（默认为最后一个）
popped = fruits.pop()
print("pop后:", fruits)
print("被弹出的元素:", popped)

# index() - 返回指定元素的索引
index = fruits.index("草莓")
print("草莓的索引:", index)

# count() - 返回指定元素的出现次数
numbers = [1, 2, 3, 2, 4, 2, 5]
count = numbers.count(2)
print("2出现的次数:", count)

# sort() - 对列表进行排序（原地排序）
numbers.sort()
print("排序后:", numbers)

# reverse() - 反转列表
numbers.reverse()
print("反转后:", numbers)

# copy() - 复制列表
fruits_copy = fruits.copy()
print("复制的列表:", fruits_copy)

# clear() - 清空列表
fruits.clear()
print("清空后:", fruits)

```
## 2. 元组
**def:有序、不可变、允许重复元素的集合**
### 2.1 创建元组
```py
# 创建空元组
empty_tuple = ()
print(empty_tuple)

# 创建包含元素的元组
numbers = (1, 2, 3, 4, 5)
fruits = ("苹果", "香蕉", "橙子")
mixed = (1, "Hello", 3.14, True)
print(numbers)
print(fruits)
print(mixed)

# 创建只包含一个元素的元组（需要加逗号）
single_element = (1,)
print(single_element)
print(type(single_element))

# 不使用括号创建元组
numbers = 1, 2, 3, 4, 5
print(numbers)
print(type(numbers))

# 使用tuple()函数创建元组
string_tuple = tuple("Hello")
range_tuple = tuple(range(5))
print(string_tuple)
print(range_tuple)
```

### 2.2 访问元组元素
```py
fruits = ("苹果", "香蕉", "橙子", "葡萄", "西瓜")

# 使用索引访问元素（索引从0开始）
print(fruits[0])  # 第一个元素
print(fruits[2])  # 第三个元素

# 使用负数索引（从末尾开始计数）
print(fruits[-1])  # 最后一个元素
print(fruits[-2])  # 倒数第二个元素

# 切片（获取子元组）
print(fruits[1:3])    # 索引1到2的元素
print(fruits[:3])     # 从开始到索引2的元素
print(fruits[2:])     # 从索引2到末尾的元素
print(fruits[1:5:2])  # 索引1到4，步长为2的元素
print(fruits[::-1])   # 反转元组
```

### 2.3 元组操作
```py
tuple1 = (1, 2, 3)
tuple2 = (4, 5, 6)

# 连接元组
combined = tuple1 + tuple2
```

```py

print("香蕉的索引:", index)
```

## 3. 字典
**def:无序、可变、键值对（key-value）的集合**
### 3.1 创建字典
```py
# 创建空字典
empty_dict = {}
print(empty_dict)

# 创建包含键值对的字典
student = {
    "name": "张三",
    "age": 20,
    "gender": "男",
    "courses": ["数学", "物理", "化学"]
}
print(student)

# 使用dict()函数创建字典
dict1 = dict(name="李四", age=21, gender="女")
print(dict1)

# 从键值对序列创建字典
keys = ["a", "b", "c"]
values = [1, 2, 3]
dict2 = dict(zip(keys, values))
print(dict2)
```

### 3.2 访问字典元素
```py
student = {
    "name": "张三",
    "age": 20,
    "gender": "男",
    "courses": ["数学", "物理", "化学"]
}

# 使用键访问值
print(student["name"])
print(student["courses"])

# 使用get()方法访问值（如果键不存在，返回None或指定的默认值）
print(student.get("age"))
print(student.get("grade"))  # 键不存在，返回None
print(student.get("grade", "未知"))  # 键不存在，返回默认值"未知"

# 获取所有键
keys = student.keys()
print("所有键:", keys)

# 获取所有值
values = student.values()
print("所有值:", values)

# 获取所有键值对
items = student.items()
print("所有键值对:", items)
```

### 3.3 修改字典
```py
student = {
    "name": "张三",
    "age": 20,
    "gender": "男"
}
print("原始字典:", student)

# 添加或修改键值对
student["age"] = 21  # 修改已存在的键
student["grade"] = "大三"  # 添加新的键
print("修改后:", student)

# 使用update()方法更新字典
student.update({"age": 22, "courses": ["数学", "物理"]})
print("update后:", student)
```

### 3.4 删除字典元素
```py
student = {
    "name": "张三",
    "age": 20,
    "gender": "男",
    "grade": "大三"
}
print("原始字典:", student)

# 使用del删除键值对
del student["grade"]
print("del后:", student)

# 使用pop()删除并返回指定键的值
age = student.pop("age")
print("pop后:", student)
print("被删除的值:", age)

# 使用popitem()删除并返回最后一个键值对
item = student.popitem()
print("popitem后:", student)
print("被删除的键值对:", item)

# 使用clear()清空字典
student.clear()
print("clear后:", student)

```
### 3.5 字典遍历
```py
student = {
    "name": "张三",
    "age": 20,
    "gender": "男",
    "grade": "大三"
}

# 遍历键
print("遍历键:")
for key in student:
    print(key)

# 遍历值
print("\n遍历值:")
for value in student.values():
    print(value)

# 遍历键值对
print("\n遍历键值对:")
for key, value in student.items():
    print(f"{key}: {value}")
```
