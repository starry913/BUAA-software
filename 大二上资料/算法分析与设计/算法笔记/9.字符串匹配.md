# 字符串
## 1.KMP算法
用于匹配两个字符串<br>
这个算法要求字符串从1开始<br>
**核心思想：文本串指针i永不后退，只调整模式串指针j**<br>
- i：文本串当前位置，只向前移动，从不回退（这是KMP相比暴力算法的优势）<br>
- j：模式串已匹配的长度，通过next数组调整，实现模式串的"滑动"<br>
- 当失配时，利用已匹配信息（next数组），将模式串向右滑动，而不是将文本串指针回退<br>
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>
#include <ctype.h>
char p[10005],s[10005];//p为模式串，s为文本串 
int nex[10005];
//nex[i]表示[1~i]字符串的最长真前后缀长度
//也表示[1~i]模式串下标失配时应该移动到的位置 
int main()
{
	scanf("%s",p+1);	int m=strlen(p+1);
	scanf("%s",s+1);	int n=strlen(s+1);
	
	//get next
	nex[0]=nex[1]=0;
	for(int i=2,j=0;i<=m;i++)
	{
		while(j&&p[i]!=p[j+1])	j=nex[j];
		if(p[i]==p[j+1])	j++;
		nex[i]=j;
	}
	
	//match
	//KMP核心思想：文本串指针i永不后退，只通过调整模式串指针j来实现"滑动"
	//i表示文本串当前位置，j表示模式串已匹配的长度
	for(int i=1,j=0;i<=n;i++)
	{
		//失配时，调整j（模式串位置），而不是回退i（文本串位置）
		//利用next数组，将模式串向右滑动，避免重复比较
		while(j&&s[i]!=p[j+1])	j=nex[j];
		//匹配成功，j向前移动
		if(s[i]==p[j+1])	j++;
		if(j==m)
		{
			//匹配成功，匹配位置为 i-m+1
			//do what you want! 
		}
	}
    return 0;
}
```
## 2.字符串Hash
将字符串转为一个ull整数
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>
#include <ctype.h>
const int N=1e6+9;
char s[N],p[N];
typedef unsigned long long ull;
const ull base=131;
ull	h1[N],h2[N],b[N];
//h[i]表示s[1~i]的hash（类似带阶数的前缀和）
//b[i]表示base^i
ull getHash(ull h[],int l,int r)//利用前缀和算子串[l~r]的hash 
{
	return h[r]-h[l-1]*b[r-l+1];
}
ull getHash2(ull h[],int l,int r)//利用后缀和算逆字串[r~l]的hash 
{
	return h[l]-h[r+1]*b[r-l+1];
}

int main()
{
	scanf("%s",p+1);	int m=strlen(p+1);
	scanf("%s",s+1);	int n=strlen(s+1);
	b[0]=1;
	for(int i=1;i<=n;i++)
	{
		b[i]=b[i-1]*base;
		h1[i]=h1[i-1]*base+(int)p[i];
		h2[i]=h2[i-1]*base+(int)s[i];
        //h[n+1-i]=h[n+2-i]*base+(int)s[n+1-i];    //后缀和求法
	}
	for(int i=1;i+m-1<=n;i++)
	{
		if(getHash(h1,1,m)==getHash(h2,i,i+m-1))
		{
			//匹配成功，do what you want!
		}
	}
    return 0;
}
```

### 3.1 对字符串增加或删去字符，使得s变为一个回文串
求最小代价
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>
#include <ctype.h>
char s[5000005];
int cost[300];
int dp[2025][2025];//dp[i][j]表示i~j的字符串变成回文串的最小代价 
int min(int a,int b)
{
	if(a<b)	return a;
	return b;
}
int main()
{
	int m,n,len,x,y;
	char c;
	scanf("%d%d",&m,&n);
	scanf("%s",s+1);
	len=strlen(s+1);
	while(m--)
	{
		scanf(" %c%d%d",&c,&x,&y);
		if(x<y)	cost[c]=x;
		else	cost[c]=y;
	}
	int l,r;
	for(r=1;r<=len;r++)
	{
		for(l=r-1;l>=1;l--)
		{
			if(s[l]==s[r]) dp[l][r]=dp[l+1][r-1];
			else
			{
				dp[l][r]=min(dp[l+1][r]+cost[s[l]],dp[l][r-1]+cost[s[r]]);
			}
		}
	} 
	printf("%d",dp[1][len]);
    return 0;
}
```

## 4.字典树
用来求字符串t在多个字符串s1~sn中是否出现过以及出现次数
![alt text](360截图20250707153634705.jpg)
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>
#include <ctype.h>
int nex[N][27]; //nex[i][0]表示从i点出发，边为a的下一个节点地址
int cnt[N];     //cnt[i]表示以i节点结尾的字符串数量
int idx=2;      //内存池，用于动态开点
void insert(char s[])
{
    int n=strlen(s);
    int x=1;
    for(int i=0;i<=n-1;i++)
    {
        if(!nex[x][s[i]-'a']) nex[x][s[i]-'a']=idx++;
		x=nex[x][s[i]-'a'];
    }
    cnt[x]++;
}
int check(char s[])
{
	int n=strlen(s);
	int x=1;
	for(int i=0;i<=n-1;i++) x=nex[x][s[i]-'a'];
	return cnt[x];
}

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
	{
		char s[N];
		scanf("%s",s);
		insert(s);        //插入字典树
	}
	for(i=1;i<=m;i++)
	{
		char s[N];
		scanf("%s",s);
		printf("%d\n",check(s));    //查找字典树
	}

    return 0;
}
```

例题1
求字符串公共前缀
![alt text](360截图20250826144749838.jpg)
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

long long nex[1000005][30];
int idx=1;
long long sum[1000005];
long long num[1000005];
long long dep[1000005];
char s[10000005];
const long long mod=998244353;

void insert(char s[])
{
    int x=1;
    int n=strlen(s);
    for(int i=0;i<n;i++)
    {
        sum[x]++;
        if(!nex[x][s[i]-'a'])
        {
            nex[x][s[i]-'a']=++idx;
            dep[idx]=dep[x]+1;
        }
        x=nex[x][s[i]-'a'];
        
    }
    sum[x]++;
    num[x]++;
}

long long solve()
{
    long long ans=0;
    for(int x=1;x<=idx;x++)
    {
        //情况1:ab和ab
        if(num[x]>=1)
        {
            ans+=(num[x]*num[x])%mod*dep[x]%mod;
            ans%=mod;
        }
        
        //情况2:abc和abf
        int tmp=0;
        for(int i=0;i<26;i++)
        {
            int y=nex[x][i];
            if(y==0) continue;
            ans+=2*tmp*sum[y]%mod*dep[x]%mod;
            ans%=mod;
            tmp+=sum[y];
            tmp%=mod;
        }
        
        //情况3:ab和abf
        ans+=2*num[x]*tmp%mod*dep[x]%mod;
        ans%=mod;
    }
    return ans;
}

int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%s",s);
        insert(s);
    }
    printf("%lld",solve()%mod);
    return 0;

}
```
## 5.有限状态机(FSM)
用于匹配文本串和模式串<br>
状态：0表示初始状态，1~m表示匹配到第几个字符，m表示匹配成功<br>
**状态转移表构建原理：**<br>
- 利用模式串自身的前缀后缀匹配关系（类似KMP的next数组）<br>
- 当在状态i失配时，找到pattern[0...i-1]的最长公共前后缀对应的状态x<br>
- 从状态x继续匹配，避免重复比较已匹配的部分<br>
- 例如：模式串"ababc"，状态3时已匹配"aba"，失配时回退到状态1（因为"a"是"aba"的最长公共前后缀）<br>
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

char text[10005], pattern[10005];
int trans[10005][128];  // trans[state][char] = next_state
int state_arr[10005];  // state_arr[i]表示文本串第i位为结尾的匹配状态（匹配到模式串的哪个位置）

// 构建状态转移表
// 核心思想：利用模式串自身的前缀后缀匹配关系
// 当在状态i失配时，找到最长真前缀=真后缀的位置x，从x状态继续匹配
void buildFSM(char pattern[], int m)
{
    // 初始化：所有状态遇到所有字符都回到0
    for(int i = 0; i <= m; i++)
    {
        for(int j = 0; j < 128; j++)
        {
            trans[i][j] = 0;
        }
    }
    
    // 状态0遇到pattern[0]转移到状态1
    trans[0][pattern[0]] = 1;
    
    // 构建状态转移：使用影子状态x（类似KMP的next数组思想）
    // x表示状态i的最长真前缀=真后缀对应的状态（即模式串pattern[0...i-1]的最长公共前后缀长度）
    int x = 0;  // 初始时，状态1的前缀后缀匹配长度为0
    for(int i = 1; i < m; i++)
    {
        // 关键：复制状态x的所有转移
        // 因为pattern[0...x-1]是pattern[0...i-1]的最长公共前后缀
        // 所以当在状态i失配时，可以回退到状态x继续匹配
        for(int j = 0; j < 128; j++)
        {
            trans[i][j] = trans[x][j];
        }
        // 匹配成功转移到下一状态
        trans[i][pattern[i]] = i + 1;
        // 更新影子状态x：x = trans[x][pattern[i]]
        // 这相当于找到pattern[0...i]的最长公共前后缀对应的状态
        x = trans[x][pattern[i]];
    }
}

// 使用状态机匹配
int matchFSM(char text[], int n, int m)
{
    int state = 0;
    int count = 0;
    
    for(int i = 0; i < n; i++)
    {
        state = trans[state][text[i]];
        state_arr[i] = state;  // 记录第i位为结尾的匹配状态
        
        if(state == m)  // 匹配成功
        {
            count++;
            int pos = i - m + 1;  // 匹配位置（从0开始）
            printf("匹配位置: %d\n", pos);
            state = trans[0][text[i]];  // 继续匹配
            state_arr[i] = state;  // 更新状态
        }
    }
    
    return count;
}

int main()
{
    scanf("%s", pattern);
    scanf("%s", text);
    
    int m = strlen(pattern);
    int n = strlen(text);
    
    buildFSM(pattern, m);
    int result = matchFSM(text, n, m);
    printf("匹配次数: %d\n", result);
    
    // 输出每个位置的状态
    printf("每个位置的状态值:\n");
    for(int i = 0; i < n; i++)
    {
        printf("位置%d: 状态值=%d", i, state_arr[i]);
        if(state_arr[i] == m) printf(" (匹配成功)");
        printf("\n");
    }
    
    return 0;
}
```

```cpp
#include <bits/stdc++.h>

int main()
{
    string s;
    int count=0;
    while(getline(cin,s))
    {
        transform(s.begin(), s.end(), s.begin(), ::tolower);
        stringstream ss(s);
        string word;
        int flag=0;
        while (ss >> word)
        {
            if(word.find("cac") == 0)
            {
                flag=1;
                break;
            }
            else if(word.length() >= 3 && word.rfind("tus") == word.length() - 3)
            {
                flag=1;
                break;
            }
            else if(word.find("cactus") != string::npos)
            {
                flag=1;
                break;
            }
        }
        if(flag==1) count++;
    }

    return 0;
}
```
