# 数学
## 快速傅里叶变换(FFT)

**使用说明**：这是一个封装好的FFT板子，每个函数都有清晰的输入输出说明，适用于大多数FFT相关题目。

```c
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

#define PI 3.14159265358979323846

// ==================== 数据结构 ====================
// 复数结构体
typedef struct {
    double real;  // 实部
    double imag;  // 虚部
} Complex;

// ==================== 复数基本运算 ====================
// 这些函数不需要理解原理，直接使用即可

// 复数加法
Complex cadd(Complex a, Complex b) {
    Complex c;
    c.real = a.real + b.real;
    c.imag = a.imag + b.imag;
    return c;
}

// 复数减法
Complex csub(Complex a, Complex b) {
    Complex c;
    c.real = a.real - b.real;
    c.imag = a.imag - b.imag;
    return c;
}

// 复数乘法
Complex cmul(Complex a, Complex b) {
    Complex c;
    c.real = a.real * b.real - a.imag * b.imag;
    c.imag = a.real * b.imag + a.imag * b.real;
    return c;
}

// ==================== FFT核心函数 ====================

/**
 * 位逆序置换（内部函数，FFT需要）
 * 输入: a[] - 复数数组, n - 数组长度（必须是2的幂）
 * 输出: 原地修改数组，按位逆序排列
 * 用途: FFT算法的预处理步骤，不需要单独调用
 */
void bitReverse(Complex *a, int n) {
    int i, j, k;
    for (i = 1, j = n >> 1; i < n - 1; i++) {
        if (i < j) {
            Complex temp = a[i];
            a[i] = a[j];
            a[j] = temp;
        }
        k = n >> 1;
        while (j >= k) {
            j -= k;
            k >>= 1;
        }
        j += k;
    }
}

/**
 * FFT主函数（核心函数）
 * 输入: 
 *   - a[]: 复数数组，存储多项式的系数（实部）或点值（结果）
 *   - n: 数组长度，必须是2的幂（如2, 4, 8, 16, ...）
 *   - inv: 0表示正变换(FFT)，1表示逆变换(IFFT)
 * 输出: 
 *   - 原地修改数组a[]
 *   - 如果inv=0: a[k] = 多项式在单位根ω^k处的值
 *   - 如果inv=1: a[k] = 多项式的第k个系数（需要先做FFT，再点值相乘）
 * 用途: 
 *   - 正变换：将系数表示转换为点值表示
 *   - 逆变换：将点值表示转换为系数表示
 * 
 * 使用示例:
 *   Complex a[8];
 *   // 初始化a[]为多项式系数
 *   fft(a, 8, 0);  // 正变换，得到点值
 *   // 现在a[k] = 多项式在ω^k处的值
 */
void fft(Complex *a, int n, int inv) {
    bitReverse(a, n);
    int i, j;
    for (int len = 2; len <= n; len <<= 1) {
        double ang = 2 * PI / len * (inv ? -1 : 1);
        Complex wlen;
        wlen.real = cos(ang);
        wlen.imag = sin(ang);
        for (i = 0; i < n; i += len) {
            Complex w;
            w.real = 1.0;
            w.imag = 0.0;
            for (j = 0; j < len / 2; j++) {
                Complex u = a[i + j];
                Complex v = cmul(a[i + j + len / 2], w);
                a[i + j] = cadd(u, v);
                a[i + j + len / 2] = csub(u, v);
                w = cmul(w, wlen);
            }
        }
    }
    if (inv) {
        for (i = 0; i < n; i++) {
            a[i].real /= n;
            a[i].imag /= n;
        }
    }
}

// ==================== 常用功能函数 ====================

/**
 * 计算两个多项式的乘积
 * 输入:
 *   - a[]: 第一个多项式的系数（整数数组）
 *   - n1: 第一个多项式的项数（次数+1）
 *   - b[]: 第二个多项式的系数（整数数组）
 *   - n2: 第二个多项式的项数（次数+1）
 *   - c[]: 输出数组，存储结果多项式的系数
 * 输出:
 *   - c[]: 乘积多项式的系数，c[i]是x^i的系数
 *   - 结果多项式的项数 = n1 + n2 - 1
 * 用途: 计算两个多项式的乘积，时间复杂度O(n log n)
 * 
 * 使用示例:
 *   int a[] = {1, 2, 3};  // 1 + 2x + 3x^2
 *   int b[] = {4, 5};      // 4 + 5x
 *   int c[10];
 *   polyMultiply(a, 3, b, 2, c);
 *   // 结果: c = {4, 13, 22, 15}  // 4 + 13x + 22x^2 + 15x^3
 */
void polyMultiply(int *a, int n1, int *b, int n2, int *c) {
    // 计算需要的FFT长度（必须是2的幂）
    int n = 1;
    while (n < n1 + n2 - 1) n <<= 1;
    
    // 分配内存
    Complex *fa = (Complex*)malloc(n * sizeof(Complex));
    Complex *fb = (Complex*)malloc(n * sizeof(Complex));
    
    // 初始化：将整数系数转换为复数
    for (int i = 0; i < n; i++) {
        fa[i].real = (i < n1) ? a[i] : 0;
        fa[i].imag = 0;
        fb[i].real = (i < n2) ? b[i] : 0;
        fb[i].imag = 0;
    }
    
    // 步骤1: 对两个多项式做FFT（系数→点值）
    fft(fa, n, 0);
    fft(fb, n, 0);
    
    // 步骤2: 点值相乘
    for (int i = 0; i < n; i++) {
        fa[i] = cmul(fa[i], fb[i]);
    }
    
    // 步骤3: IFFT（点值→系数）
    fft(fa, n, 1);
    
    // 步骤4: 提取结果（四舍五入）
    for (int i = 0; i < n1 + n2 - 1; i++) {
        c[i] = (int)(fa[i].real + 0.5);
    }
    
    free(fa);
    free(fb);
}

/**
 * 计算多项式在单位根处的值（DFT）
 * 输入:
 *   - coeff[]: 多项式的系数（整数数组）
 *   - n_coeff: 系数的个数
 *   - result[]: 输出数组，存储结果
 *   - n_result: 结果数组长度（必须是2的幂，且>=n_coeff）
 * 输出:
 *   - result[k] = 多项式在单位根ω^k处的值（复数）
 * 用途: 计算多项式在所有单位根处的值，用于需要点值的题目
 * 
 * 使用示例:
 *   int coeff[] = {1, 2, 3, 4};
 *   Complex result[4];
 *   polyDFT(coeff, 4, result, 4);
 *   // 现在result[k] = 多项式在ω^k处的值
 */
void polyDFT(int *coeff, int n_coeff, Complex *result, int n_result) {
    // 初始化：将系数转换为复数
    for (int i = 0; i < n_result; i++) {
        result[i].real = (i < n_coeff) ? coeff[i] : 0;
        result[i].imag = 0;
    }
    
    // FFT：系数→点值
    fft(result, n_result, 0);
    
    // 现在result[k]就是多项式在ω^k处的值
}

// ==================== 使用示例 ====================
int a[1000005],b[1000005],c[2000010];
// 示例1: 多项式乘法
void example1_polyMultiply() {
    int n,m;
    scanf("%d%d",&n,&m);//n是a的次数，m是b的次数
    for(int i=0;i<=n;i++) scanf("%d",&a[i]);
    for(int i=0;i<=m;i++) scanf("%d",&b[i]);
    
    polyMultiply(a, n+1, b, m+1, c);//这里放的是项数，项数=次数+1
    
    for (int i = 0; i <= n+m; i++) {
        printf("%d ",c[i]);
    }
    // 输出: 4, 13, 22, 15
}

// 示例2: 计算多项式在单位根处的值
void example2_polyDFT() {
    int coeff[] = {1, 2, 3, 4};
    Complex result[4];
    
    polyDFT(coeff, 4, result, 4);
    
    printf("多项式在单位根处的值：\n");
    for (int i = 0; i < 4; i++) {
        printf("f(ω^%d) = %.2f + %.2fi\n", 
               i, result[i].real, result[i].imag);
    }
}

// 示例3: 解决"计算加权和"类型的题目
void example3_weightedSum() {
    int n;
    scanf("%d", &n);
    int size = 1 << n;  // 2^n
    
    int *coeff = (int*)malloc(size * sizeof(int));
    for (int i = 0; i < size; i++) {
        scanf("%d", &coeff[i]);
    }
    
    // 计算多项式在单位根处的值
    Complex *values = (Complex*)malloc(size * sizeof(Complex));
    polyDFT(coeff, size, values, size);
    
    // 计算加权和 z = Σ cos(k) · f(ω^k)
    Complex z;
    z.real = 0.0;
    z.imag = 0.0;
    for (int k = 0; k < size; k++) {
        double cos_k = cos(k);
        Complex term;
        term.real = cos_k * values[k].real;
        term.imag = cos_k * values[k].imag;
        z = cadd(z, term);
    }
    
    printf("%.2f %.2f\n", z.real, z.imag);
    
    free(coeff);
    free(values);
}

}
```

### 板子使用指南

**1. 什么时候用哪个函数？**

| 题目类型 | 使用的函数 | 说明 |
|---------|----------|------|
| 多项式乘法 | `polyMultiply()` | 直接调用，输入两个多项式系数，输出乘积系数 |
| 需要点值 | `polyDFT()` | 计算多项式在单位根处的值 |
| 自定义操作 | `fft()` | 需要自己处理点值相乘等操作 |

**2. 常见问题**

**Q: n必须是2的幂吗？**
A: 是的。如果多项式有m个系数，需要找到最小的2的幂n，使得n≥m，然后补0。

**Q: 如何补0？**
A: 在调用FFT前，将数组后面补0即可。`polyMultiply()`和`polyDFT()`已经自动处理了。

**Q: 精度问题？**
A: 结果可能有浮点误差，使用`(int)(x + 0.5)`四舍五入。

**Q: 内存管理？**
A: `polyMultiply()`和`polyDFT()`内部会分配内存，使用完后会自动释放。如果直接使用`fft()`，需要自己管理内存。

**3. 快速上手**

```c
// 最简单的用法：多项式乘法
int a[] = {1, 2, 3};
int b[] = {4, 5};
int c[10];
polyMultiply(a, 3, b, 2, c);
// 完成！c[]就是结果
```
