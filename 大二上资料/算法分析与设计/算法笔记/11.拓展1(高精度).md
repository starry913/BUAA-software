# 高精度
## 1.fft高精度乘法
```c
#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include<string.h>

#define PI 3.14159265358979323846

// ==================== 数据结构 ====================
// 复数结构体
typedef struct {
    double real;  // 实部
    double imag;  // 虚部
} Complex;

// ==================== 复数基本运算 ====================
// 这些函数不需要理解原理，直接使用即可

// 复数加法
Complex cadd(Complex a, Complex b) {
    Complex c;
    c.real = a.real + b.real;
    c.imag = a.imag + b.imag;
    return c;
}

// 复数减法
Complex csub(Complex a, Complex b) {
    Complex c;
    c.real = a.real - b.real;
    c.imag = a.imag - b.imag;
    return c;
}

// 复数乘法
Complex cmul(Complex a, Complex b) {
    Complex c;
    c.real = a.real * b.real - a.imag * b.imag;
    c.imag = a.real * b.imag + a.imag * b.real;
    return c;
}

void bitReverse(Complex *a, int n) {
    int i, j, k;
    for (i = 1, j = n >> 1; i < n - 1; i++) {
        if (i < j) {
            Complex temp = a[i];
            a[i] = a[j];
            a[j] = temp;
        }
        k = n >> 1;
        while (j >= k) {
            j -= k;
            k >>= 1;
        }
        j += k;
    }
}

void fft(Complex *a, int n, int inv) {
    bitReverse(a, n);
    int i, j;
    for (int len = 2; len <= n; len <<= 1) {
        double ang = 2 * PI / len * (inv ? -1 : 1);
        Complex wlen;
        wlen.real = cos(ang);
        wlen.imag = sin(ang);
        for (i = 0; i < n; i += len) {
            Complex w;
            w.real = 1.0;
            w.imag = 0.0;
            for (j = 0; j < len / 2; j++) {
                Complex u = a[i + j];
                Complex v = cmul(a[i + j + len / 2], w);
                a[i + j] = cadd(u, v);
                a[i + j + len / 2] = csub(u, v);
                w = cmul(w, wlen);
            }
        }
    }
    if (inv) {
        for (i = 0; i < n; i++) {
            a[i].real /= n;
            a[i].imag /= n;
        }
    }
}

void polyMultiply(int *a, int n1, int *b, int n2, int *c) {
    // 计算需要的FFT长度（必须是2的幂）
    int n = 1;
    while (n < n1 + n2 - 1) n <<= 1;
    
    // 分配内存
    Complex *fa = (Complex*)malloc(n * sizeof(Complex));
    Complex *fb = (Complex*)malloc(n * sizeof(Complex));
    
    // 初始化：将整数系数转换为复数
    for (int i = 0; i < n; i++) {
        fa[i].real = (i < n1) ? a[i] : 0;
        fa[i].imag = 0;
        fb[i].real = (i < n2) ? b[i] : 0;
        fb[i].imag = 0;
    }
    
    // 步骤1: 对两个多项式做FFT（系数→点值）
    fft(fa, n, 0);
    fft(fb, n, 0);
    
    // 步骤2: 点值相乘
    for (int i = 0; i < n; i++) {
        fa[i] = cmul(fa[i], fb[i]);
    }
    
    // 步骤3: IFFT（点值→系数）
    fft(fa, n, 1);
    
    // 步骤4: 提取结果（四舍五入）
    for (int i = 0; i < n1 + n2 - 1; i++) {
        c[i] = (int)(fa[i].real + 0.5);
    }
    
    free(fa);
    free(fb);
}

char s[1000005];
int a[1000005],b[1000005],c[2000010];
int main() 
{
    int n,m,t;
    scanf("%d",&t);
    while(t--)
    {
        memset(a,0,sizeof(a));
        memset(b,0,sizeof(b));
        memset(c,0,sizeof(c));
        memset(s,0,sizeof(s));

        scanf("%s",s);
        n=strlen(s);
        n--;
        for(int i=n;i>=0;i--)
        {
            a[n-i]=s[i]-'0';
        }

        memset(s,0,sizeof(s));
    
        scanf("%s",s);
        m=strlen(s);
        m--;
        for(int i=m;i>=0;i--)
        {
            b[m-i]=s[i]-'0';
        }
        polyMultiply(a, n+1, b, m+1, c);//这里放的是项数，项数=次数+1
        
        // 处理进位，结果数组最大索引是n+m+1（因为两个长度为n+1和m+1的多项式相乘，结果长度最多是n+m+2）
        int i=0,k=0;
        int maxLen = n + m; // 最大可能的次数
        while(i <= maxLen || k > 0)
        {
            if(i > maxLen) maxLen = i;
            c[i] += k;
            k = c[i] / 10;
            c[i] = c[i] % 10;
            i++;
        }
        
        // 输出结果，跳过前导0
        int flag = 0;
        for(int i = maxLen ; i >= 0; i--)
        {
            if(c[i] == 0 && flag == 0) continue;
            if(c[i] != 0) flag = 1;
            printf("%d", c[i]);
        }
        // 如果结果是0，需要输出0
        if(flag == 0) printf("0");
        printf("\n");
    }
    return 0;
}
```
