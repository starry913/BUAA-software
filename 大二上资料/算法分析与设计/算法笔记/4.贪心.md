# 贪心
## 1.课程选择
有n门课，每门课时间为[a,b],问最多能上几门课
思路:按结束时间升序排列
```c
#include <stdio.h>
#include <stdlib.h>
int pt[500005][2];

int x_ascending(const void * pt1, const void * pt2);

int main()
{
    int n,i;
    scanf("%d",&n);
	for(i=0;i<=n-1;i++)
	{
		scanf("%d%d",&pt[i][0],&pt[i][1]); 
	}
    qsort(pt, n, 2 * sizeof(int), x_ascending);
    int t=pt[0][1];
    int count=1;
    for(i=1;i<=n-1;i++)
	{
		if(pt[i][0]>=t)
		{
			t=pt[i][1];
			count++;
		}
	}
	printf("%d",count);
    return 0;
}

int x_ascending(const void * pt1, const void * pt2)//前降序后升序
{
if(((int*) pt1)[1] < ((int *) pt2)[1]) return -1;
else if(((int *) pt1)[1] > ((int *) pt2)[1]) return 1;

else if (((int *) pt1)[0] < ((int *) pt2)[0]) return -1;
else if(((int *) pt1)[0] > ((int *) pt2)[0]) return 1;

else return 0;
}
```
