# 动态规划
## 1.矩阵相乘

**使用说明**：计算n个矩阵相乘的最少乘法次数。输入p数组，p[i]表示第i个矩阵的行数，p[i+1]表示第i个矩阵的列数（也是第i+1个矩阵的行数）。

**矩阵维度说明**：矩阵 Ai 的行数为 pi，列数为 pi+1。

**状态转移方程**：`m[i][j] = min(i≤k<j) (m[i][k] + m[k+1][j] + p[i] * p[k+1] * p[j+1])`

```c
#include <stdio.h>
#include <limits.h>

#define MAXN 305
#define INF (1LL << 60)  // 足够大的值，避免溢出

int p[MAXN];              // p[i]表示矩阵Ai的行数，p[i+1]表示矩阵Ai的列数
long long m[MAXN][MAXN];  // m[i][j]表示从矩阵Ai到矩阵Aj相乘的最少乘法次数
int s[MAXN][MAXN];        // s[i][j]记录最优分割点k，用于输出括号方案

// 递归输出括号方案
void printParenthesis(int i, int j) {
    if (i == j) {
        printf("A%d", i);
    } else {
        printf("(");
        printParenthesis(i, s[i][j]);
        printParenthesis(s[i][j] + 1, j);
        printf(")");
    }
}

int main() {
    int n;
    scanf("%d", &n);  // n个矩阵
    
    // 输入p数组：p[1]到p[n+1]，共n+1个数
    // 矩阵Ai的行数为p[i]，列数为p[i+1]
    for (int i = 1; i <= n + 1; i++) {
        scanf("%d", &p[i]);
    }
    
    // 初始化：单个矩阵不需要乘法
    for (int i = 1; i <= n; i++) {
        m[i][i] = 0;
    }
    
    // 动态规划：按长度从小到大计算
    // len表示矩阵链的长度（包含的矩阵个数）
    for (int len = 2; len <= n; len++) 
    {
        // i表示起始矩阵编号
        for (int i = 1; i <= n - len + 1; i++) 
        {
            int j = i + len - 1;  // j表示结束矩阵编号
            m[i][j] = INF;
            
            // 尝试所有可能的分割点k
            // 将矩阵链[Ai...Aj]分割为[Ai...Ak]和[Ak+1...Aj]
            // 左边结果矩阵维度：p[i] × p[k+1]
            // 右边结果矩阵维度：p[k+1] × p[j+1]
            // 合并需要：p[i] * p[k+1] * p[j+1] 次乘法
            for (int k = i; k < j; k++) 
            {
                long long cost = m[i][k] + m[k+1][j] + (long long)p[i] * p[k+1] * p[j+1];
                if (cost < m[i][j]) {
                    m[i][j] = cost;
                    s[i][j] = k;  // 记录最优分割点
                }
            }
        }
    }
    
    // 输出结果：m[1][n]表示从矩阵A1到矩阵An相乘的最少乘法次数
    printf("最少乘法次数: %lld\n", m[1][n]);
    
    // 输出括号方案
    printf("最优括号方案: ");
    printParenthesis(1, n);
    printf("\n");
    
    return 0;
}
```

**示例**：
- 输入：3个矩阵，p[1..4] = [10, 20, 30, 40]
  - 矩阵A1：p[1]×p[2] = 10×20
  - 矩阵A2：p[2]×p[3] = 20×30
  - 矩阵A3：p[3]×p[4] = 30×40
- 输出：最少乘法次数

**时间复杂度**：O(n³)
**空间复杂度**：O(n²)

## 2.最优二叉搜索树(OBST)

**使用说明**：给定n个关键字和它们的访问频率，构建一棵期望搜索成本最小的二叉搜索树。

**问题描述**：
- 有n个关键字 k1, k2, ..., kn，每个关键字ki的访问概率为p[i]
- 有n+1个虚拟键（失败查找）d0, d1, ..., dn，每个虚拟键di的访问概率为q[i]
- 构建一棵二叉搜索树，使得期望搜索成本最小

**状态转移方程**：
- `w[i][j] = w[i][j-1] + p[j] + q[j]` （概率总和）
- `e[i][j] = min(i≤r≤j) (e[i][r-1] + e[r+1][j] + w[i][j])`

其中：
- `e[i][j]`：从关键字i到j构建的最优二叉搜索树的期望搜索成本
- `w[i][j]`：从关键字i到j（包括虚拟键）的概率总和
- `root[i][j]`：记录最优根节点

```c
#include <stdio.h>
#include <limits.h>

#define MAXN 1005
#define INF INT_MAX

double p[MAXN];        // p[i]表示关键字ki的访问概率
double q[MAXN];        // q[i]表示虚拟键di的访问概率
double e[MAXN][MAXN];  // e[i][j]表示从关键字i到j的最优期望搜索成本
double w[MAXN][MAXN];  // w[i][j]表示从关键字i到j的概率总和
int root[MAXN][MAXN];  // root[i][j]记录最优根节点
int leftChild[MAXN];    // leftChild[i]表示节点i的左儿子
int rightChild[MAXN];   // rightChild[i]表示节点i的右儿子

// 递归构建树结构，返回根节点
int buildTree(int i, int j) {
    if (i > j) {
        return -1;  // 空子树
    }
    
    int r = root[i][j];
    
    // 构建左子树
    if (i <= r - 1) {
        leftChild[r] = buildTree(i, r - 1);
    } else {
        leftChild[r] = -1;  // 左子树为空
    }
    
    // 构建右子树
    if (r + 1 <= j) {
        rightChild[r] = buildTree(r + 1, j);
    } else {
        rightChild[r] = -1;  // 右子树为空
    }
    
    return r;
}

int main() {
    int n;
    scanf("%d", &n);  // n个关键字
    
    // 输入关键字访问概率 p[1..n]
    for (int i = 1; i <= n; i++) {
        scanf("%lf", &p[i]);
    }
    
    // 输入虚拟键访问概率 q[0..n]
    for (int i = 0; i <= n; i++) {
        scanf("%lf", &q[i]);
    }
    
    // 初始化：单个虚拟键的情况
    for (int i = 0; i <= n; i++) {
        e[i+1][i] = q[i];  // 只有虚拟键di
        w[i+1][i] = q[i];
    }
    
    // 动态规划：按长度从小到大计算
    // len表示关键字的个数
    for (int len = 1; len <= n; len++) 
    {
        // i表示起始关键字编号
        for (int i = 1; i <= n - len + 1; i++) 
        {
            int j = i + len - 1;  // j表示结束关键字编号
            e[i][j] = INF;
            w[i][j] = w[i][j-1] + p[j] + q[j];  // 计算概率总和
            
            // 尝试所有可能的根节点r
            // 将树分为左子树[i..r-1]和右子树[r+1..j]
            for (int r = i; r <= j; r++) 
            {
                double cost = e[i][r-1] + e[r+1][j] + w[i][j];
                if (cost < e[i][j]) {
                    e[i][j] = cost;
                    root[i][j] = r;  // 记录最优根节点
                }
            }
        }
    }
    
    // 初始化左右儿子数组
    for (int i = 1; i <= n; i++) {
        leftChild[i] = -1;
        rightChild[i] = -1;
    }
    
    // 构建树结构
    buildTree(1, n);
    
    // 输出结果：第一行输出最小代价和（四舍五入到整数）
    printf("%.0f\n", e[1][n]);
    
    // 输出n行，每行两个整数，表示节点i的左儿子和右儿子
    for (int i = 1; i <= n; i++) {
        printf("%d %d\n", leftChild[i], rightChild[i]);
    }
    
    return 0;
}
```

**示例**：
- 输入：n=3个关键字
  - p[1..3] = [0.15, 0.10, 0.05] （关键字k1, k2, k3的访问概率）
  - q[0..3] = [0.05, 0.10, 0.05, 0.05] （虚拟键d0, d1, d2, d3的访问概率）
- 输出：最优期望搜索成本

**说明**：
- 关键字ki在树中的深度为d(ki)，访问成本为(d(ki) + 1)
- 虚拟键di在树中的深度为d(di)，访问成本为d(di)
- 期望搜索成本 = Σ(p[i] * (d(ki) + 1)) + Σ(q[i] * d(di))

**时间复杂度**：O(n³)
**空间复杂度**：O(n²)

**优化版本（Knuth优化）**：
可以使用四边形不等式优化到O(n²)，但标准版本已经足够使用。

```c
#include<stdio.h>
int a[10000005],dp[10000005];
int t=1;
//dp[t]表示t个不降序列的末尾元素最小值 
int main() 
{
    int n,i,j,ans=0;
    scanf("%d",&n);
    for(i=1;i<=n;i++)    scanf("%d",&a[i]);
    dp[1]=a[1];
    for(i=2;i<=n;i++)
    {
		if(a[i]>=dp[t])
		{
			t++;
			dp[t]=a[i];
		}
		else if(a[i]<dp[t])
		{
			for(j=t;j>=0;j--)
			{
				if(a[i]>=dp[j])
				{
					dp[j+1]=a[i]; 
                    break;
				}
			}
		}
    }
    printf("%d",t);
    return 0; 
}
```
## 4.最长公共子序列(LCS)
只有朴素法$O(n*m)$<br>
dp[i][j]表示a[0~i-1], b[0~j-1]中最长公共子序列长度<br>

```c
#include <stdio.h>
#include <string.h>
char a[10005],b[10005];
int dp[10005][10005];
int max(int a,int b)
{
    if(a>b) return a;
    return b;
}
int main()
{
    int i,j,ans,n,m;
    scanf("%s",a);
    scanf("%s",b);
    n=strlen(a);
    m=strlen(b);
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
        {
            if(a[i-1]==b[j-1]) dp[i][j]=dp[i-1][j-1]+1;
            else           dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
        }
    ans=dp[n][m];
    //求具体序列
    int x=n,y=m;
    char stack[6000];//也可以改成char型,是一个栈
    int top=-1;
    while(x!=0&&y!=0)
    {
        if(a[x-1]==b[y-1])
        {
            top++;
            stack[top]=a[x-1];
            x--,y--;
        } 
        else if(dp[x-1][y]>dp[x][y-1])    x--;
        else    y--;   
    }
    for(i=top;i>=0;i--) printf("%c",stack[i]);
    printf("\n%d\n",ans);
    return 0;
}
```

## 5.最长公共子串(LCS)
dp[i][j]表示以a[i-1],b[j-1]结尾的最长公共子串的长度<br>
```c
#include <stdio.h>
#include <string.h>
char a[10005],b[10005];
int dp[10005][10005];
int max(int a,int b)
{
    if(a>b) return a;
    return b;
}
int main()
{
    int i,j,n,m,max1=0;
    scanf("%s",a);
    scanf("%s",b);
    n=strlen(a);
    m=strlen(b);
    // 先计算所有dp值
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
        {
            if(a[i-1]==b[j-1]) 
            {
                dp[i][j]=dp[i-1][j-1]+1;
            }
            else dp[i][j]=0;
        }
    
    // 单独遍历dp数组找最大值和位置
    int max_i=0,max_j=0;
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
        {
            if(dp[i][j]>max1)
            {
                max1=dp[i][j];
                max_i=i;
                max_j=j;
            }
        }
    
    printf("%d ",max1);
    
    // 输出具体的最长公共子串
    if(max1>0)
    {
        char result[10005];
        int idx=0;
        // 从结束位置向前回溯max1个字符
        for(i=max_i-max1;i<max_i;i++)
        {
            result[idx]=a[i];
            idx++;
        }
        result[idx]='\0';
        printf("%s\n",result);
    }
    else
    {
        printf("\n");
    }
    return 0;
}
```

## 6.钢条切割问题

**使用说明**：给定一根长度为n的钢条和一个价格表，求如何切割钢条使得总收益最大。

**问题描述**：
- 有一根长度为n的钢条
- 价格表p[i]表示长度为i的钢条的价格（i=1,2,...,n）
- 可以将钢条切割成多段，求最大收益

**状态转移方程**：
- `dp[i] = max(1≤j≤i) (p[j] + dp[i-j])`
- `dp[i]`：长度为i的钢条的最大收益

**方法1：自底向上动态规划（推荐）**

```c
#include <stdio.h>
#define MAXN 100005

long long p[MAXN];  // p[i]表示长度为i的钢条的价格
long long dp[MAXN]; // dp[i]表示长度为i的钢条的最大收益
long long s[MAXN];  // s[i]记录长度为i的钢条的最优切割方案（第一段长度）

int main() {
    int n;
    scanf("%d", &n);  // 钢条长度
    
    // 输入价格表：p[1]到p[n]
    for (int i = 1; i <= n; i++) {
        scanf("%lld", &p[i]);
    }
    
    // 初始化
    dp[0] = 0;  // 长度为0的钢条收益为0
    
    // 动态规划：自底向上计算
    for (int i = 1; i <= n; i++) {
        dp[i] = p[i];  // 初始化为不切割的情况（直接卖整根）
        s[i] = i;      // 初始化为不切割
        // 尝试所有可能的第一段长度j
        for (int j = 1; j < i; j++) {
            long long cost = p[j] + dp[i - j];
            if (cost > dp[i]) {
                dp[i] = cost;
                s[i] = j;  // 记录最优切割方案
            }
        }
    }
    
    // 输出最大收益
    printf("%lld\n", dp[n]);
    
    long long stack[MAXN];
    int top = -1,count=0;
    // 输出最优切割方案
    int len = n;
    while (len > 0) {
        top++;
        stack[top]=s[len];
        len -= s[len];
        count++;
    }
    printf("%d\n",count);
    for(int i=0;i<=top;i++) printf("%lld ",stack[i]);
    
    return 0;
}
```

## 7.流水线调度

```c
#include <stdio.h>

    else return c;

}
```

### 8.2 完全背包(每个物品有无穷个)
```c
//O(N*V)
scanf("%d%d",&N,&V);    //N个物品，背包容量为V
for(i=1;i<=N;i++)
{
    scanf("%d%d",&v,&w); //每个物品的v体积和w价值
    for(j=0;j<=V;j++)    //从前往后遍历
    {
        if(j>=v)dp[j]=max(dp[j],dp[j-v]+w);
    }    
}
ans=dp[V];
```

### 8.3 多重背包(每个物品有S个)
```c
//朴素法O(N*V*S)
scanf("%d%d",&N,&V);    //N个物品，背包容量为V
for(i=1;i<=N;i++)
{
    scanf("%d%d%d",&v,&w,&s); //每个物品的v体积、w价值、s个数
    while(s--)
    {
        for(j=V;j>=0;j--)    //从后往前遍历
        {
            if(j>=v)dp[j]=max(dp[j],dp[j-v]+w);
        }   
    } 
}
ans=dp[V];

//二进制优化模型O(N*V*log(S))
//s=1+2+4+8+……+剩余的非2进制的数
//14=1+2+4+7
scanf("%d%d",&N,&V);    //N个物品，背包容量为V
for(i=1;i<=N;i++)
{
    scanf("%d%d%d",&v,&w,&s); //每个物品的v体积、w价值、s个数
    //s=14
    for(k=1;k<=s;s=s-k,k=k*2)
    {
        //k=1,2,4
        for(j=V;j>=0;j--)    //从后往前遍历
        {
            if(j>=k*v)dp[j]=max(dp[j],dp[j-k*v]+k*w);
        }   
    }
    //s=6
    for(j=V;j>=0;j--)    //从后往前遍历
    {
        if(j>=s*v)dp[j]=max(dp[j],dp[j-s*v]+s*w);
    }  
}
ans=dp[V];
```

### 8.6 巨型01背包(V极大，会TLE)
已改为按价值 DP，避免按容量遍历。
 O(N × 总价值)，而不是 O(N × V)
```c
#include <stdio.h>
long long dp[1000005];//dp[j] 表示价值达到j时的最小体积
long long min(long long a,long long b)
{
    return a<b? a:b;
}
int main()
{
    long long N,V,i,j,v,w,ans=0;
    long long sum_value=0;
    scanf("%lld%lld",&N,&V);
    
    for(i=0;i<=1000000;i++) dp[i]=1e18;
    dp[0]=0;
    
    for(i=1;i<=N;i++)
    {
        scanf("%lld%lld",&v,&w);
        sum_value+=w;
        for(j=sum_value;j>=w;j--)
        {
            if(dp[j-w]!=1e18)
            {
                dp[j]=min(dp[j],dp[j-w]+v);
            }
        }
    }
    
    for(i=sum_value;i>=0;i--)
    {
        if(dp[i]<=V)
        {
            ans=i;
            break;
        }
    }
    printf("%lld",ans);
    return 0;
}
```
